## New Architecture

### Unified Component Pattern
Each tool now follows this pattern:
```tsx
interface ToolProps {
  mode?: 'card' | 'modal'
  // ... other props
  onClose?: () => void
  onCancel?: () => void
}

export function Tool({ mode = 'card', ...props }: ToolProps) {
  if (mode === 'card') {
    return <ToolCardWrapper>...</ToolCardWrapper>
  }
  
  return <Dialog>...</Dialog> // Modal mode
}
```
